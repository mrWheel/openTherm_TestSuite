NEOPIXEL Issue:
The pin configuration (_NEOPIXEL_PIN=48) is correct for the built-in NeoPixel on ESP32-S3-DevKitC.

Test this code modification in blinkNeopixels():

void blinkNeopixels()
{
  if (millis() > neopixelTimer)
  {
    neopixelTimer = millis() + DELAY_NOPIXELS;
    strip.setBrightness(255);  // Full brightness
    strip.setPixelColor(0, strip.Color(255, 0, 0));  // Bright RED
    strip.show();
    delay(300);
    strip.setPixelColor(0, strip.Color(0, 0, 0));    // OFF
    strip.show();
  }
}
